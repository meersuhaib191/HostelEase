        CASE WHEN a.day2 = 'P' THEN 1 ELSE 0 END +
        CASE WHEN a.day3 = 'P' THEN 1 ELSE 0 END +

        CASE WHEN a.day31 = 'P' THEN 1 ELSE 0 END) AS days_present,
    'unpaid' AS status  -- Default status as unpaid
FROM userregistration u
JOIN attendance_details a ON u.enrollment_no = a.enrollment_no
GROUP BY u.enrollment_no";

$stmt = $mysqli->prepare($query);
$stmt->execute();
$studentsResult = $stmt->get_result();
$students = [];

// Fetch the student details and calculate total bill
while ($row = $studentsResult->fetch_assoc()) {
    $totalBill = $row['days_present'] * $billPerDay;  // Calculate total bill based on attendance and daily rate
    $row['total_bill'] = number_format($totalBill, 2); // Format the total bill for output
    $students[] = $row;
}

echo json_encode(['status' => 'success', 'students' => $students, 'billPerDay' => $billPerDay]);

<script>
let studentsData = []; // Store student data to prevent data loss on refresh

function calculateBill() {
    $.ajax({
        url: 'calculate_bill.php',
        type: 'GET',
        dataType: 'json',
        success: function(response) {
            if (response.status === 'success') {
                const billPerDay = response.billPerDay;
                studentsData = response.students; // Store student data
                updateTable(studentsData, billPerDay);

                // Show the modal after table update
                $('#billModal').modal('show');
            } else {
                alert("Error calculating bill. Please try again.");
            }
        },
        error: function() {
            alert("Error occurred during calculation. Please check your server or try again.");
        }
    });
}

function updateTable(students, billPerDay) {
    const tableBody = $("#studentTableBody");
    tableBody.empty();

    students.forEach((student, index) => {
        const totalBill = (student.days_present * billPerDay).toFixed(2);
        const row = `<tr>
            <td>${index + 1}</td>
            <td>${student.firstName} ${student.lastName}</td>
            <td>${student.days_present}</td>
            <td>₹${billPerDay.toFixed(2)}</td>
            <td>₹${totalBill}</td>
            <td>
                <select class="status-dropdown" onchange="updateStatus(${index})">
                    <option value="Paid" ${student.status === 'Paid' ? 'selected' : ''}>Paid</option>
                    <option value="Unpaid" ${student.status === 'Unpaid' ? 'selected' : ''}>Unpaid</option>
                </select>
            </td>
        </tr>`;
        tableBody.append(row);
    });
}

function updateStatus(index) {
    studentsData[index].status = $(".status-dropdown")[index].value;
}

function sendBillEmails() {
    $.ajax({
        url: 'send_bill_emails.php',
        type: 'POST',
        success: function(response) {
            alert("Emails sent successfully!");
        },
        error: function() {
            alert("Error sending emails.");
        }
    });
}

function downloadPDF() {
    window.location.href = 'generate_pdf.php';
}

// Filter table by search
function filterTable() {
    var input, filter, table, tr, td, i, j, txtValue;
    input = document.getElementById("searchBox");
    filter = input.value.toUpperCase();
    table = document.getElementById("studentTable");
    tr = table.getElementsByTagName("tr");

    for (i = 1; i < tr.length; i++) {
        tr[i].style.display = "none";
        td = tr[i].getElementsByTagName("td");
        for (j = 0; j < td.length; j++) {
            if (td[j]) {
                txtValue = td[j].textContent || td[j].innerText;
                if (txtValue.toUpperCase().indexOf(filter) > -1) {
                    tr[i].style.display = "";
                    break;
                }
            }
        }
    }
}
